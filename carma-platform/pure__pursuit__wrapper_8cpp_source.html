43void PurePursuitWrapper::trajectoryPlanHandler(const cav_msgs::TrajectoryPlan::ConstPtr& tp)
44{
45 ROS_DEBUG_STREAM("Received TrajectoryPlanCurrentPosecallback message");
46
47 std::vector<double> times;
48 std::vector<double> downtracks;
49
50 std::vector<cav_msgs::TrajectoryPlanPoint> trajectory_points = tp->trajectory_points;
51
52 ROS_DEBUG_STREAM("Original Trajectory size:"<<trajectory_points.size());
53
54
55 trajectory_utils::conversions::trajectory_to_downtrack_time(trajectory_points, &downtracks, &times);
56
57 //detect stopping case
58 size_t stopping_index = 0;
59 for (size_t i = 1; i < times.size(); i++)
60 {
61 if (times[i] == times[i - 1]) //if exactly same, it is stopping case
62 {
63 ROS_DEBUG_STREAM("Detected a stopping case where times is exactly equal: " << times[i-1]);
64 ROS_DEBUG_STREAM("And index of that is: " << i << ", where size is: " << times.size());
65 stopping_index = i;
66 break;
67 }
68 }
69
70 std::vector<double> speeds;
71 trajectory_utils::conversions::time_to_speed(downtracks, times, tp->initial_longitudinal_velocity, &speeds);
72
73 if (speeds.size() != trajectory_points.size())
74 {
75 throw std::invalid_argument("Speeds and trajectory points sizes do not match");
76 }
77
78 for (size_t i = 0; i < speeds.size(); i++) { // Ensure 0 is min speed
79 if (stopping_index != 0 && i >= stopping_index - 1)
80 {
81 speeds[i] = 0.0; //stopping case
82 }
83 else
84 {
85 speeds[i] = std::max(0.0, speeds[i]);
86 }
87 }
88
89 std::vector<double> lag_speeds = apply_response_lag(speeds, downtracks, config_.vehicle_response_lag); // This call requires that the first speed point be current speed to work as expected
90
91 autoware_msgs::Lane lane;
92 lane.header = tp->header;
93 std::vector<autoware_msgs::Waypoint> waypoints;
94 waypoints.reserve(trajectory_points.size());
95
96 for (int i = 0; i < trajectory_points.size(); i++)
97 {
98 autoware_msgs::Waypoint wp;
99
100 wp.pose.pose.position.x = trajectory_points[i].x;
101 wp.pose.pose.position.y = trajectory_points[i].y;
102 wp.twist.twist.linear.x = lag_speeds[i];
103 ROS_DEBUG_STREAM("Setting waypoint idx: " << i <<", with planner: << " << trajectory_points[i].planner_plugin_name << ", x: " << trajectory_points[i].x <<
104 ", y: " << trajectory_points[i].y <<
105 ", speed: " << lag_speeds[i]* 2.23694 << "mph");
106 waypoints.push_back(wp);
107 }
108
109 lane.waypoints = waypoints;
110
111 waypoint_pub_(lane);
112
113};

116std::vector<double> PurePursuitWrapper::apply_response_lag(const std::vector<double>& speeds, const std::vector<double> downtracks, double response_lag) const { // Note first speed is assumed to be vehicle speed
117 if (speeds.size() != downtracks.size()) {
118 throw std::invalid_argument("Speed list and downtrack list are not the same size.");
119 }
120
121 std::vector<double> output;
122 if (speeds.empty()) {
123 return output;
124 }
125
126 double lookahead_distance = speeds[0] * response_lag;
127
128 double downtrack_cutoff = downtracks[0] + lookahead_distance;
129 size_t lookahead_count = std::lower_bound(downtracks.begin(),downtracks.end(), downtrack_cutoff) - downtracks.begin(); // Use binary search to find lower bound cutoff point
130 output = trajectory_utils::shift_by_lookahead(speeds, (unsigned int) lookahead_count);
131 return output;
132}

134std::vector<cav_msgs::TrajectoryPlanPoint> PurePursuitWrapper::remove_repeated_timestamps(const std::vector<cav_msgs::TrajectoryPlanPoint>& traj_points){
135
136 std::vector<cav_msgs::TrajectoryPlanPoint> new_traj_points;
137
138 cav_msgs::TrajectoryPlanPoint prev_point;
139 bool first = true;
140
141 for(auto point : traj_points){
142
143 if(first){
144 first = false;
145 prev_point = point;
146 new_traj_points.push_back(point);
147 continue;
148 }
149
150 if(point.target_time != prev_point.target_time){
151 new_traj_points.push_back(point);
152 prev_point = point;
153 }
154 else{
155 ROS_DEBUG_STREAM("Duplicate point found");
156 }
157 }
158
159 return new_traj_points;
160
161}
