43 carma_ros2_utils::CallbackReturn Node::handle_on_configure(const rclcpp_lifecycle::State &)
44 {
45 // Reset config
46 config_ = Config();
47
48 // Load parameters
49 get_parameter<int>("num_of_entries", config_.num_of_entries);
50
51 // Register runtime parameter update callback
52 add_on_set_parameters_callback(std::bind(&Node::parameter_update_callback, this, std_ph::_1));
53
54 // Setup subscribers
55 mo_sub_ = create_subscription<carma_v2x_msgs::msg::MobilityOperation>("mobility_operation_inbound", 5,
56 std::bind(&Node::mobilityOperationCallback, this, std_ph::_1));
57
58 // Setup publishers
59 mr_pub_ = create_publisher<carma_v2x_msgs::msg::MobilityRequest>("mobility_request_outbound", 5);
60 cav_detection_pub_ = create_publisher<std_msgs::msg::String>("cav_truck_identified", 5);
61 content_pub_ = create_publisher<std_msgs::msg::String>("truck_safety_info", 5);
62
63 // Setup service servers
64 inspection_request_service_server_ = create_service<std_srvs::srv::Trigger>("send_inspection_request",std::bind(&Node::inspectionRequestCallback, this, std_ph::_1, std_ph::_2, std_ph::_3));
65
66
67 content_pub_timer_ = create_timer(
68 get_clock(),
69 std::chrono::milliseconds(50),
70 std::bind(&Node::content_pub_timer_callback, this));
71
72 RCLCPP_INFO_STREAM(this->get_logger(), "Truck inspection plugin is initialized...");
73
74 // Return success if everthing initialized successfully
75 return CallbackReturn::SUCCESS;
76 }

100 void Node::mobilityOperationCallback(carma_v2x_msgs::msg::MobilityOperation::UniquePtr msg)
101 {
102 // if there is a truck around running CARMA
103 if(msg->strategy == Node::INSPECTION_STRATEGY)
104 {
105 // if the incoming message contains a valid safety log
106 if(isSafetyLogValid(msg->strategy_params))
107 {
108 safety_log_ = msg->strategy_params;
109 safety_log_ += ",timestamp:" + std::to_string(msg->m_header.timestamp);
110 } else {
111 std_msgs::msg::String msg_out;
112 std::string k_v_pair = msg->strategy_params;
113 // get only VIN number, state and license plate
114 msg_out.data = k_v_pair;
115 // publish message to show there is a cav truck in the radio range of the ego-vehicle
116 cav_detection_pub_->publish(msg_out);
117 }
118 }
119 }

122 bool Node::isSafetyLogValid(const std::string& log)
123 {
124 // Check 1: if number of k-v pairs matches expectation
125 std::string temp;
126 int count = 0;
127 std::stringstream ss(log);
128 while(std::getline(ss, temp, ','))
129 {
130 if(!temp.empty())
131 {
132 count++;
133 }
134 else
135 {
136 // do not allow spaces in the safety log
137 return false;
138 }
139 }
140 return count == this->number_of_entries;
141 }
