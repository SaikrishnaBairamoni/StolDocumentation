8Introduction
9============
10
11This Sphinx plugin can be used to filter out warnings that are known to be false
12positives. The warnings are filtered out based on a set of regular expressions
13given via an configuration file. The format of the configuration file is a
14plain-text file where each line consists of a regular expression. Any lines
15starting with ``#`` will be ignored.
16
17Configuration options
18=====================
19
20- ``warnings_filter_config``: Configuration file.
21- ``warnings_filter_silent``: Silent flag. If True, warning is hidden. If False
22 the warning is converted to an information message and displayed.
23"""

37 """Warnings filter.

50 def filter(self, record: logging.LogRecord) -> bool:
51 if record.levelno != logging.WARNING:
52 return True
53
54 for expression in self._expressions:
55 if re.match(expression, record.msg):
56 if self._silent:
57 return False
58 else:
59 record.levelno = logging.INFO

66def configure(app: Sphinx) -> None:
67 """Entry point.
68
69 Args:
70 app: Sphinx application instance.
71 """
72
73 # load expressions from configuration file
74 with open(app.config.warnings_filter_config) as f:
75 expressions = list()
76 for line in f.readlines():
77 if not line.startswith("#"):
78 expressions.append(line.rstrip())
79
80 # install warnings filter to all the Sphinx logger handlers
81 filter = WarningsFilter(expressions, app.config.warnings_filter_silent)
82 logger = logging.getLogger(NAMESPACE)
83 for handler in logger.handlers:
84 handler.filters.insert(0, filter)
