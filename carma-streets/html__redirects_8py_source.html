19# Mechanism to generate static HTML redirect pages in the output
20#
21# Uses redirect_template.html and the list of pages given in
22# conf.html_redirect_pages
23#
24# Adapted from ideas in https://tech.signavio.com/2017/managing-sphinx-redirects
25import os.path
26
27from sphinx.builders.html import StandaloneHTMLBuilder
28
29REDIRECT_TEMPLATE = r"""
30<html>
31 <head>
32 <meta http-equiv="refresh" content="0; url=$NEWURL" />
33 <script>
34 var id=window.location.href.split("#")[1];
35
36 if (id && (/^[a-zA-Z\:\/0-9\_\-\.]+$/.test(id))) {
37 window.location.href = "$NEWURL"+"#"+id;
38 }
39 else {
40 window.location.href = "$NEWURL";
41 };
42 </script>
43 </head>
44 <body>
45 <p>Page has moved <a href="$NEWURL">here</a>.</p>
46 </body>
47</html>
48"""

51def setup(app):
52 app.add_config_value('html_redirect_pages', [], 'html')
53 app.connect('build-finished', create_redirect_pages)
54
55 # Since we're just setting up a build-finished hook, which runs
56 # after both reading and writing, this extension is safe for both.
57 return {
58 'parallel_read_safe': True,
59 'parallel_write_safe': True,
60 }

63def create_redirect_pages(app, docname):
64 if not isinstance(app.builder, StandaloneHTMLBuilder):
65 return # only relevant for standalone HTML output
66
67 for (old_url, new_url) in app.config.html_redirect_pages:
68 if old_url.startswith('/'):
69 old_url = old_url[1:]
70
71 new_url = app.builder.get_relative_uri(old_url, new_url)
72 out_file = app.builder.get_outfilename(old_url)
73
74 out_dir = os.path.dirname(out_file)
75 if not os.path.exists(out_dir):
76 os.makedirs(out_dir)
77
78 content = REDIRECT_TEMPLATE.replace("$NEWURL", new_url)
79
80 if not os.path.exists(out_file):
81 with open(out_file, "w") as rp:
82 rp.write(content)
