14# This could be as simple as generating a couple of sets of instructions, one

16class ZephyrAppCommandsDirective(Directive):
17 r'''
18 This is a Zephyr directive for generating consistent documentation
19 of the shell commands needed to manage (build, flash, etc.) an application.
20
21 For example, to generate commands to build samples/hello_world for
22 qemu_x86 use::
23
24 .. zephyr-app-commands::
25 :zephyr-app: samples/hello_world
26 :board: qemu_x86
27 :goals: build
28
29 Directive options:
30
31 \:tool:
32 which tool to use. Valid options are currently 'cmake', 'west' and 'all'.
33 The default is 'west'.
34
35 \:app:
36 path to the application to build.
37
38 \:zephyr-app:
39 path to the application to build, this is an app present in the upstream
40 zephyr repository. Mutually exclusive with \:app:.
41
42 \:cd-into:
43 if set, build instructions are given from within the \:app: folder,
44 instead of outside of it.
45
46 \:generator:
47 which build system to generate. Valid options are
48 currently 'ninja' and 'make'. The default is 'ninja'. This option
49 is not case sensitive.
50
51 \:host-os:
52 which host OS the instructions are for. Valid options are
53 'unix', 'win' and 'all'. The default is 'all'.
54
55 \:board:
56 if set, the application build will target the given board.
57
58 \:shield:
59 if set, the application build will target the given shield.
60
61 \:conf:
62 if set, the application build will use the given configuration
63 file. If multiple conf files are provided, enclose the
64 space-separated list of files with quotes, e.g., "a.conf b.conf".
65
66 \:gen-args:
67 if set, additional arguments to the CMake invocation
68
69 \:build-args:
70 if set, additional arguments to the build invocation
71
72 \:build-dir:
73 if set, the application build directory will *APPEND* this
74 (relative, Unix-separated) path to the standard build directory. This is
75 mostly useful for distinguishing builds for one application within a
76 single page.
77
78 \:goals:
79 a whitespace-separated list of what to do with the app (in
80 'build', 'flash', 'debug', 'debugserver', 'run'). Commands to accomplish
81 these tasks will be generated in the right order.
82
83 \:maybe-skip-config:
84 if set, this indicates the reader may have already
85 created a build directory and changed there, and will tweak the text to
86 note that doing so again is not necessary.
87
88 \:compact:
89 if set, the generated output is a single code block with no
90 additional comment lines
91
92 \:west-args:
93 if set, additional arguments to the west invocation (ignored for CMake)
94
95 '''

124 def run(self):
125 # Re-run on the current document if this directive's source changes.
126 self.state.document.settings.env.note_dependency(__file__)
127
128 # Parse directive options. Don't use os.path.sep or os.path.join here!
129 # That would break if building the docs on Windows.
130 tool = self.options.get('tool', 'west').lower()
131 app = self.options.get('app', None)
132 zephyr_app = self.options.get('zephyr-app', None)
133 cd_into = 'cd-into' in self.options
134 generator = self.options.get('generator', 'ninja').lower()
135 host_os = self.options.get('host-os', 'all').lower()
136 board = self.options.get('board', None)
137 shield = self.options.get('shield', None)
138 conf = self.options.get('conf', None)
139 gen_args = self.options.get('gen-args', None)
140 build_args = self.options.get('build-args', None)
141 build_dir_append = self.options.get('build-dir', '').strip('/')
142 goals = self.options.get('goals').split()
143 skip_config = 'maybe-skip-config' in self.options
144 compact = 'compact' in self.options
145 west_args = self.options.get('west-args', None)
146
147 if tool not in self.TOOLS:
148 raise self.error('Unknown tool {}; choose from: {}'.format(
149 tool, self.TOOLS))
150
151 if app and zephyr_app:
152 raise self.error('Both app and zephyr-app options were given.')
153
154 if generator not in self.GENERATORS:
155 raise self.error('Unknown generator {}; choose from: {}'.format(
156 generator, self.GENERATORS))
157
158 if host_os not in self.HOST_OS:
159 raise self.error('Unknown host-os {}; choose from: {}'.format(
160 host_os, self.HOST_OS))
161
162 if compact and skip_config:
163 raise self.error('Both compact and maybe-skip-config options were given.')
164
165 app = app or zephyr_app
166 in_tree = self.IN_TREE_STR if zephyr_app else None
167 # Allow build directories which are nested.
168 build_dir = ('build' + '/' + build_dir_append).rstrip('/')
169
170 # Create host_os array
171 host_os = [host_os] if host_os != "all" else [v for v in self.HOST_OS
172 if v != 'all']
173 # Create tools array
174 tools = [tool] if tool != "all" else [v for v in self.TOOLS
175 if v != 'all']
176 # Build the command content as a list, then convert to string.
177 content = []
178 tool_comment = None
179 if len(tools) > 1:
180 tool_comment = 'Using {}:'
181
182 run_config = {
183 'host_os': host_os,
184 'app': app,
185 'in_tree': in_tree,
186 'cd_into': cd_into,
187 'board': board,
188 'shield': shield,
189 'conf': conf,
190 'gen_args': gen_args,
191 'build_args': build_args,
192 'build_dir': build_dir,
193 'goals': goals,
194 'compact': compact,
195 'skip_config': skip_config,
196 'generator': generator,
197 'west_args': west_args
198 }
199
200 if 'west' in tools:
201 w = self._generate_west(**run_config)
202 if tool_comment:
203 paragraph = nodes.paragraph()
204 paragraph += nodes.Text(tool_comment.format('west'))
205 content.append(paragraph)
206 content.append(self._lit_block(w))
207 else:
208 content.extend(w)
209
210 if 'cmake' in tools:
211 c = self._generate_cmake(**run_config)
212 if tool_comment:
213 paragraph = nodes.paragraph()
214 paragraph += nodes.Text(tool_comment.format(
215 'CMake and {}'.format(generator)))
216 content.append(paragraph)
217 content.append(self._lit_block(c))
218 else:
219 content.extend(c)
220
221 if not tool_comment:
222 content = [self._lit_block(content)]
223
224 return content

226 def _lit_block(self, content):
227 content = '\n'.join(content)
228
229 # Create the nodes.
230 literal = nodes.literal_block(content, content)
231 self.add_name(literal)
232 literal['language'] = 'console'
233 return literal

236 def _generate_west(self, **kwargs):
237 content = []
238 generator = kwargs['generator']
239 board = kwargs['board']
240 app = kwargs['app']
241 in_tree = kwargs['in_tree']
242 goals = kwargs['goals']
243 cd_into = kwargs['cd_into']
244 build_dir = kwargs['build_dir']
245 compact = kwargs['compact']
246 west_args = kwargs['west_args']
247 kwargs['board'] = None
248 # west always defaults to ninja
249 gen_arg = ' -G\'Unix Makefiles\'' if generator == 'make' else ''
250 cmake_args = gen_arg + self._cmake_args(**kwargs)
251 cmake_args = ' --{}'.format(cmake_args) if cmake_args != '' else ''
252 west_args = ' {}'.format(west_args) if west_args else ''
253 # ignore zephyr_app since west needs to run within
254 # the installation. Instead rely on relative path.
255 src = ' {}'.format(app) if app and not cd_into else ''
256 dst = ' -d {}'.format(build_dir) if build_dir != 'build' else ''
257
258 if in_tree and not compact:
259 content.append(in_tree)
260
261 if cd_into and app:
262 content.append('cd {}'.format(app))
263
264 # We always have to run west build.
265 #
266 # FIXME: doing this unconditionally essentially ignores the
267 # maybe-skip-config option if set.
268 #
269 # This whole script and its users from within the
270 # documentation needs to be overhauled now that we're
271 # defaulting to west.
272 #
273 # For now, this keeps the resulting commands working.
274 content.append('west build -b {}{}{}{}{}'.
275 format(board, west_args, dst, src, cmake_args))
276
277 # If we're signing, we want to do that next, so that flashing
278 # etc. commands can use the signed file which must be created
279 # in this step.
280 if 'sign' in goals:
281 content.append('west sign{}'.format(dst))
282
283 for goal in goals:
284 if goal in {'build', 'sign'}:
285 continue
286 elif goal == 'flash':
287 content.append('west flash{}'.format(dst))
288 elif goal == 'debug':
289 content.append('west debug{}'.format(dst))
290 elif goal == 'debugserver':
291 content.append('west debugserver{}'.format(dst))
292 elif goal == 'attach':
293 content.append('west attach{}'.format(dst))
294 else:
295 content.append('west build -t {}{}'.format(goal, dst))
296
297 return content

326 def _cd_into(self, mkdir, **kwargs):
327 app = kwargs['app']
328 host_os = kwargs['host_os']
329 compact = kwargs['compact']
330 build_dir = kwargs['build_dir']
331 skip_config = kwargs['skip_config']
332 content = []
333 os_comment = None
334 if len(host_os) > 1:
335 os_comment = '# On {}'
336 num_slashes = build_dir.count('/')
337 if not app and mkdir and num_slashes == 0:
338 # When there's no app and a single level deep build dir,
339 # simplify output
340 content.extend(self._mkdir(mkdir, build_dir, 'all',
341 skip_config))
342 if not compact:
343 content.append('')
344 return content
345 for host in host_os:
346 if host == "unix":
347 if os_comment:
348 content.append(os_comment.format('Linux/macOS'))
349 if app:
350 content.append('cd {}'.format(app))
351 elif host == "win":
352 if os_comment:
353 content.append(os_comment.format('Windows'))
354 if app:
355 backslashified = app.replace('/', '\\')
356 content.append('cd {}'.format(backslashified))
357 if mkdir:
358 content.extend(self._mkdir(mkdir, build_dir, host, skip_config))
359 if not compact:
360 content.append('')
361 return content

363 def _generate_cmake(self, **kwargs):
364 generator = kwargs['generator']
365 cd_into = kwargs['cd_into']
366 app = kwargs['app']
367 in_tree = kwargs['in_tree']
368 build_dir = kwargs['build_dir']
369 build_args = kwargs['build_args']
370 skip_config = kwargs['skip_config']
371 goals = kwargs['goals']
372 compact = kwargs['compact']
373
374 content = []
375
376 if in_tree and not compact:
377 content.append(in_tree)
378
379 if cd_into:
380 num_slashes = build_dir.count('/')
381 mkdir = 'mkdir' if num_slashes == 0 else 'mkdir -p'
382 content.extend(self._cd_into(mkdir, **kwargs))
383 # Prepare cmake/ninja/make variables
384 source_dir = ' ' + '/'.join(['..' for i in range(num_slashes + 1)])
385 cmake_build_dir = ''
386 tool_build_dir = ''
387 else:
388 source_dir = ' {}'.format(app) if app else ' .'
389 cmake_build_dir = ' -B{}'.format(build_dir)
390 tool_build_dir = ' -C{}'.format(build_dir)
391
392 # Now generate the actual cmake and make/ninja commands
393 gen_arg = ' -GNinja' if generator == 'ninja' else ''
394 build_args = ' {}'.format(build_args) if build_args else ''
395 cmake_args = self._cmake_args(**kwargs)
396
397 if not compact:
398 if not cd_into and skip_config:
399 content.append("# If you already ran cmake with -B{}, you " \
400 "can skip this step and run {} directly.".
401 format(build_dir, generator)) # noqa: E501
402 else:
403 content.append('# Use cmake to configure a {}-based build' \
404 'system:'.format(generator.capitalize())) # noqa: E501
405
406 content.append('cmake{}{}{}{}'.format(cmake_build_dir, gen_arg,
407 cmake_args, source_dir))
408 if not compact:
409 content.extend(['',
410 '# Now run ninja on the generated build system:'])
411
412 if 'build' in goals:
413 content.append('{}{}{}'.format(generator, tool_build_dir,
414 build_args))
415 for goal in goals:
416 if goal == 'build':
417 continue
418 content.append('{}{} {}'.format(generator, tool_build_dir, goal))
419
420 return content
